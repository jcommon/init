49  		super(false, StringUtil.empty, StringUtil.empty, StringUtil.empty, Title, Description);

52  		this.libraryName = libraryName;
53  	}

56  	@Override
57  	protected boolean loadSettings(final String fullResourceName, final IResourcePackage pkg, final XPath xpath, final Node node, final Document document, final IVariableProcessor varproc, final ResourceProcessorFactory factory) throws XPathException {
58  		this.path = stringAttributeValue(varproc, StringUtil.empty, node, ATTRIBUTE_PATH);
59  		this.libraryName = stringAttributeValue(varproc, StringUtil.empty, node, ATTRIBUTE_LIBRARY_NAME);
60  		
61  		return true;
62  	}
63  
64  	@Override
65  	protected boolean processResource(final String fullResourceName, final IResourcePackage pkg, final IResourceFilter filter, final IResourceProgressListener progress) {
66  		if (StringUtil.isNullOrEmpty(libraryName))
67  			return true;
68  
69  		final String dir = StringUtil.isNullOrEmpty(path) || ".".equalsIgnoreCase(path) ? jcommon.core.Process.queryWorkingDirectory() : path;
70  		if (StringUtil.isNullOrEmpty(dir))
71  			return true;
72  
73  		final File d = new File(dir);
74  		if (!d.exists())
75  			return false;
76  
77  		NativeLibrary.addSearchPath(libraryName, path);
78  		return true;
79  	}
