64  	@Override
65  	protected boolean loadSettings(final String fullResourceName, final IResourcePackage pkg, final XPath xpath, final Node node, final Document document, final IVariableProcessor varproc, final ResourceProcessorFactory factory) throws XPathException {
66  		this.name = stringAttributeValue(varproc, StringUtil.empty, node, ATTRIBUTE_NAME);
67  		this.value = stringAttributeValue(varproc, StringUtil.empty, node, ATTRIBUTE_VALUE);
68  		this.separator = stringAttributeValue(varproc, Path.pathSeparator, node, ATTRIBUTE_SEPARATOR);
69  		
70  		this.unset = boolAttributeValue(varproc, false, node, ATTRIBUTE_UNSET);
71  		this.asFile = boolAttributeValue(varproc, false, node, ATTRIBUTE_ASFILE);
72  		this.prepend = boolAttributeValue(varproc, false, node, ATTRIBUTE_PREPEND);
73  		this.append = boolAttributeValue(varproc, false, node, ATTRIBUTE_APPEND);
74  
75  		return true;

78  	@Override
79  	protected boolean processResource(final String fullResourceName, final IResourcePackage pkg, final IResourceFilter filter, final IResourceProgressListener progress) {
80  		if (StringUtil.isNullOrEmpty(name))
81  			return true;
82  
83  		String existingValue = Sys.queryEnvironmentVariable(name);
84  		if (existingValue == null)
85  			existingValue = StringUtil.empty;
86  
87  		String preppedValue;
88  		if (!asFile)
89  			preppedValue = value;
90  		else
91  			preppedValue = new File(value).getAbsolutePath();
92  
93  		String destValue;
94  		if (prepend && append)
95  			destValue = preppedValue + separator + existingValue + separator + preppedValue;
96  		else if (prepend && !append)
97  			destValue = preppedValue + separator + existingValue;
98  		else if (!prepend && append)
99  			destValue = existingValue + separator + preppedValue;
100 		else
101 			destValue = preppedValue;
102 
103 		if (!unset)
104 			return Sys.saveEnvironmentVariable(name, destValue);
105 		else
106 			return Sys.unsetEnvironmentVariable(name);
107 	}
