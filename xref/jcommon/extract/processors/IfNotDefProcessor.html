15  /**
16   * Processes or doesn't process resources depening on whether or not a variable
17   * is defined and/or has a certain value.
18   * 
19   * @author David Hoyt <dhoyt@hoytsoft.org>
20   */
21  @ResourceProcessor(
22  	tagName = "IfNotDef",
23  	supportsSize = true
24  )
25  public class IfNotDefProcessor extends ContainerProcessor {
26  	//<editor-fold defaultstate="collapsed" desc="Constants">
27  	public static final String
28  		  ATTRIBUTE_VAR         = "var"
29  		, ATTRIBUTE_VALUE       = "value"
30  		, ATTRIBUTE_IGNORECASE  = "ignoreCase"
31  	;
32  	//</editor-fold>
33  
34  	//<editor-fold defaultstate="collapsed" desc="Variables">
35  	private String var = StringUtil.empty;
36  	private String value = StringUtil.empty;
37  	private boolean ignoreCase = true;
38  	//</editor-fold>
39  
40  	//<editor-fold defaultstate="collapsed" desc="Initialization">
41  	public IfNotDefProcessor() {
42  		super();
43  	}
44  
45  	public IfNotDefProcessor(String var) {
46  		this(var, StringUtil.empty, StringUtil.empty, StringUtil.empty);
47  	}
48  
49  	public IfNotDefProcessor(String var, String Title, String Description) {
50  		this(var, StringUtil.empty, Title, Description);
51  	}
52  
53  	public IfNotDefProcessor(String var, String value, String Title, String Description) {
54  		super(Title, Description);
55  
56  		this.var = var;
57  		this.value = value;
58  	}
59  	//</editor-fold>
60  	
61  	@Override
62  	protected boolean loadContainerSettingsAfter(final String fullResourceName, final IResourcePackage pkg, final XPath xpath, final Node node, final Document document, final IVariableProcessor varproc, final ResourceProcessorFactory factory) throws XPathException {
63  		this.var = stringAttributeValue(varproc, StringUtil.empty, node, ATTRIBUTE_VAR);
64  		this.value = stringAttributeValue(varproc, null, node, ATTRIBUTE_VALUE);
65  		this.ignoreCase = boolAttributeValue(varproc, true, node, ATTRIBUTE_IGNORECASE);
66  
67  		//If we don't have a variable to examine or it doesn't exist, then there's
68  		//nothing to do here.
69  		if (StringUtil.isNullOrEmpty(var) || !varproc.hasVariable(var))
70  			return true;
71  
72  
73  		//If we don't have a value we need to compare against, then assume
74  		//that we don't want to proceess the inner resources.
75  		if (value == null)
76  			return clearProcessors();
77  
78  		//At this point, we have a value and we have a variable. Let's check if the values are equivalent.
79  
80  		final String varValue = varproc.findValue(var);
81  		if (varValue == null) 
82  			return true;
83  
84  		if (ignoreCase && !value.equalsIgnoreCase(varValue))
85  			return true;
86  
87  		if (!ignoreCase && !value.equals(varValue))
88  			return true;
89  
90  		return clearProcessors();
91  	}
