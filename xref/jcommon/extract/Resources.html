122 	protected void initFromXML(final ResourceProcessorFactory ProcessorFactory, final IVariableProcessor VariableProcessor, final InputStream XMLData) throws XPathException, ParserConfigurationException, SAXException, IOException {
123 		final DocumentBuilderFactory factory = DocumentBuilderFactory.newInstance();
124 		final DocumentBuilder builder = factory.newDocumentBuilder();
125 		final XPathFactory xpathFactory = XPathFactory.newInstance();
126 		final XPath xpath = xpathFactory.newXPath();
127 
128 		//Creates a context that always has "jcommon" as a prefix -- useful for xpath evaluations
129 		xpath.setNamespaceContext(Namespaces.createNamespaceContext());
130 
131 		this.packages = read(ProcessorFactory, VariableProcessor, xpath, builder.parse(XMLData));
132 
133 		initAfter();
134 	}

157 		//Calculate total size and resource count
158 		for(IResourcePackage pkg : packages) {
159 			if (pkg == null)
160 				continue;
161 
162 			totalResourceCount += pkg.getTotalResourceCount();
163 			totalResourceSize += pkg.getTotalSize();
164 		}

309 	public Future extract(final String[] refs, final IResourcePackage[] pkgs, final ExecutorService executor, final IResourceFilter filter, final IResourceProgressListener progress, final IResourceCallback callback) {
310 		checkRegistryInitialization();
311 
312 		synchronized(lock) {
313 			if (processed) {
314 				//If we've already processed this resource, then just mimic a
315 				//typical extraction process but don't actually do anything.
316 				if (progress == null)
317 					return CompletedFuture;
318 
319 				//Extract nothing, but make sure that
320 				return executor.submit(new Runnable() {
321 					@Override
322 					public void run() {
323 						try {
324 							//<editor-fold defaultstate="collapsed" desc="Make compiler happy">
325 							if (false)
326 								throw new InterruptedException();
327 							//</editor-fold>
328 
329 							//<editor-fold defaultstate="collapsed" desc="Prepare">
330 							if (callback != null)
331 								callback.prepare(Resources.this);
332 							//</editor-fold>
333 
334 							if (progress != null) {
335 								long startTime = System.currentTimeMillis();
336 								notifyProgressBegin(progress, startTime);
337 								notifyProgressEnd(progress, true, totalResourceSize, totalResourceCount, packages != null ? packages.length : 0, startTime, startTime);
338 							}
339 
340 							//<editor-fold defaultstate="collapsed" desc="Completed">
341 							if (callback != null)
342 								callback.completed(Resources.this);
343 							//</editor-fold>
344 						} catch(InterruptedException ie) {
345 							//<editor-fold defaultstate="collapsed" desc="Cancelled">
346 							if (callback != null)
347 								callback.cancelled(Resources.this);
348 							//</editor-fold>
349 						} catch(CancellationException ce) {
350 							//<editor-fold defaultstate="collapsed" desc="Cancelled">
351 							if (callback != null)
352 								callback.cancelled(Resources.this);
353 							//</editor-fold>
354 						} catch(RejectedExecutionException ree) {
355 							//<editor-fold defaultstate="collapsed" desc="Cancelled">
356 							if (callback != null)
357 								callback.cancelled(Resources.this);
358 							//</editor-fold>
359 						} catch(Throwable t) {
360 							//<editor-fold defaultstate="collapsed" desc="Error">
361 							if (progress != null)
362 								progress.error(t, t.getMessage());
363 							if (callback != null)
364 								callback.error(Resources.this);
365 							//</editor-fold>
366 						}
367 					}
368 				});
369 			}
370 
371 			return executor.submit(new Runnable() {
372 				@Override
373 				public void run() {
374 					synchronized(lock) {
375 						try {
376 							//<editor-fold defaultstate="collapsed" desc="Make compiler happy">
377 							if (false)
378 								throw new InterruptedException();
379 							//</editor-fold>
380 
381 							//<editor-fold defaultstate="collapsed" desc="Prepare">
382 							if (callback != null)
383 								callback.prepare(Resources.this);
384 							//</editor-fold>
385 
386 							if (!processed) {
387 								//Do the real work
388 								extractResources(refs, pkgs, filter, progress);
389 
390 								//Mark as processed
391 								processed = true;
392 							}
393 
394 							//<editor-fold defaultstate="collapsed" desc="Completed">
395 							if (callback != null)
396 								callback.completed(Resources.this);
397 							//</editor-fold>
398 						} catch(InterruptedException ie) {
399 							//<editor-fold defaultstate="collapsed" desc="Cancelled">
400 							if (callback != null)
401 								callback.cancelled(Resources.this);
402 							//</editor-fold>
403 						} catch(CancellationException ce) {
404 							//<editor-fold defaultstate="collapsed" desc="Cancelled">
405 							if (callback != null)
406 								callback.cancelled(Resources.this);
407 							//</editor-fold>
408 						} catch(RejectedExecutionException ree) {
409 							//<editor-fold defaultstate="collapsed" desc="Cancelled">
410 							if (callback != null)
411 								callback.cancelled(Resources.this);
412 							//</editor-fold>
413 						} catch(Throwable t) {
414 							//<editor-fold defaultstate="collapsed" desc="Error">
415 							if (callback != null)
416 								callback.error(Resources.this);
417 							//</editor-fold>
418 						}
419 					}
420 				}
421 			});
422 		}
423 	}

569 	protected IResourcePackage[] read(final ResourceProcessorFactory processorFactory, final IVariableProcessor variableProcessor, final XPath xpath, final Document document) throws XPathException  {
570 		Node node;
571 		NodeList lst;
572 		IResourcePackage pkg;
573 		List<IResourcePackage> pkgs = new ArrayList<IResourcePackage>(1);
574 
575 		//Collapse whitespace nodes
576 		document.normalize();
577 
578 		//Get the top-level document element, <Resources />
579 		final Element top = document.getDocumentElement();
580 
581 		//<editor-fold defaultstate="collapsed" desc="Resource name">
582 		if ((node = (Node)xpath.evaluate("//Resources", top, XPathConstants.NODE)) != null) {
583 			this.name = stringAttributeValue(variableProcessor, StringUtil.empty, node, ATTRIBUTE_NAME);
584 
585 			if (StringUtil.isNullOrEmpty(name))
586 				throw new ResourceException("Invalid resource name. It cannot be empty.");
587 
588 			this.registryReference = Registry.add(name, this);
589 		}
590 		//</editor-fold>
591 
592 		//<editor-fold defaultstate="collapsed" desc="References">
593 		//Locate <References><Add /></References> tags
594 		if ((lst = (NodeList)xpath.evaluate("//Resources/References/Add", top, XPathConstants.NODESET)) != null && lst.getLength() > 0) {
595 			final List<String> refs = new ArrayList<String>(lst.getLength());
596 
597 			//Iterate over every <Add /> tag
598 			for(int i = 0; i < lst.getLength() && (node = lst.item(i)) != null; ++i) {
599 				final String refName = stringAttributeValue(variableProcessor, StringUtil.empty, node, ATTRIBUTE_NAME);
600 				if (!StringUtil.isNullOrEmpty(refName) && !this.name.equalsIgnoreCase(refName))
601 					refs.add(refName);
602 			}
603 
604 			this.references = refs.toArray(new String[refs.size()]);
605 		}
606 		//</editor-fold>
607 
608 		//<editor-fold defaultstate="collapsed" desc="Packages">
609 		//Locate <Extract /> tags
610 		if ((lst = (NodeList)xpath.evaluate("//Resources/Extract", top, XPathConstants.NODESET)) == null || lst.getLength() <= 0)
611 			return IResourcePackage.EMPTY;
612 
613 		//Iterate over every <Extract /> tag
614 		for(int i = 0; i < lst.getLength() && (node = lst.item(i)) != null; ++i) {
615 
616 			//Ask the package to read it
617 			if ((pkg = Package.newInstance(processorFactory, variableProcessor, node, xpath, document)) != null)
618 				pkgs.add(pkg);
619 		}

630 	protected void extractResources(final int level, final String[] refs, final IResourcePackage[] pkgs, final IResourceFilter filter, final IResourceProgressListener progress) throws Throwable {
631 		//Please note that this is executed in a separate thread.
632 		//It can be cancelled or interrupted at any time.
633 
634 		int totalPkgs = 0;
635 		int totalResources = 0;
636 		long totalBytes = 0;
637 		String resourceName;
638 		long endTime;
639 		boolean success = true;
640 		Throwable exception = null;
641 		String title = null;
642 		long startTime = System.currentTimeMillis();
643 
644 		try {
645 			//Notify begin
646 			if (level == 0)
647 				notifyProgressBegin(progress, startTime);
648 
649 			//Load references
650 			if (refs != null && refs.length > 0) {
651 				for(String ref : refs) {
652 					if (ref.equalsIgnoreCase(name))
653 						continue;
654 					if (StringUtil.isNullOrEmpty(ref))
655 						throw new MissingResourceReferenceException("An empty reference name is invalid.");
656 
657 					//Attempt to locate this reference
658 					final Reference reference = Registry.findReference(ref, true);
659 					if (reference == null)
660 						continue;
661 
662 					//Get its associated resource object if it has one
663 					final Resources res = reference.getResources();
664 					if (res == null || res == this || res.isProcessed())
665 						continue;
666 
667 					//Process on this same thread
668 					res.extractResources(level + 1, res.references, res.packages, filter, progress);
669 				}
670 			}
671 
672 			for(IResourcePackage pkg : pkgs) {
673 				if (pkg == null) {
674 					notifyProgressReportPackageCompleted(progress, pkg, totalBytes, totalResources, ++totalPkgs, startTime, "Skipped package");
675 					continue;
676 				}
677 
678 				for(IResourceProcessor p : pkg) {
679 					if (p == null) {
680 						notifyProgressReportResourceCompleted(progress, p, pkg, totalBytes, ++totalResources, totalPkgs, startTime, "Skipped resource");
681 						continue;
682 					}
683 
684 					//Double check that we're allowed to process this resource
685 					resourceName = pkg.resourcePath(p.getName());
686 					if (filter != null && !filter.filter(pkg, p, resourceName)) {
687 						notifyProgressReportResourceCompleted(progress, p, pkg, totalBytes, ++totalResources, totalPkgs, startTime, "Skipped resource");
688 						continue;
689 					}
690 
691 					//Here we go!
692 					//If this was already processed, then skip the processing again
693 					if (p.isProcessed() || p.process(resourceName, pkg, filter, progress)) {
694 						totalBytes += p.getSize();
695 						title = (p instanceof DefaultResourceProcessor ? ((DefaultResourceProcessor)p).getTitle() : p.getName());
696 						if (StringUtil.isNullOrEmpty(title))
697 							title = p.getName();
698 						notifyProgressReportResourceCompleted(progress, p, pkg, totalBytes, ++totalResources, totalPkgs, startTime, title);
699 					}
700 				}
701 
702 				notifyProgressReportPackageCompleted(progress, pkg, totalBytes, totalResources, ++totalPkgs, startTime, "Completed package");
703 			}
704 			success = true;
705 		} catch(Throwable t) {
706 			success = false;
707 			exception = t;
708 		} finally {
709 			endTime = System.currentTimeMillis();
710 		}
711 
712 		if (exception != null)
713 			notifyProgressError(progress, exception, !StringUtil.isNullOrEmpty(exception.getMessage()) ? exception.getMessage() : "Error loading resources");
714 
715 		if (level == 0)
716 			notifyProgressEnd(progress, success, totalBytes, totalResources, totalPkgs, startTime, endTime);
717 	}
