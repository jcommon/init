12   * Holds variables, their values, and processes strings replacing variables with.

37  	@Override
38  	public boolean addVariables(final Map<String, String> Variables) {
39  		if (Variables == null || Variables.isEmpty())
40  			return true;
41  		for(Map.Entry<String, String> e : Variables.entrySet())
42  			vars.put(e.getKey(), e.getValue());
43  		return true;
44  	}
45  
46  	@Override
47  	public boolean saveVariable(final String Name, String Value) {
48  		if (Value == null)
49  			Value = StringUtil.empty;
50  		vars.put(Name, Matcher.quoteReplacement(Value));
51  		return true;
52  	}
53  
54  	@Override
55  	public boolean removeVariable(final String Name) {
56  		if (vars.isEmpty() || !vars.containsKey(Name))
57  			return false;
58  		vars.remove(Name);
59  		return true;
60  	}
61  
62  	@Override
63  	public boolean clearVariables() {
64  		vars.clear();
65  		return true;
66  	}
67  
68  	@Override
69  	public boolean hasVariable(final String Name) {
70  		return (vars != null && !vars.isEmpty() && vars.containsKey(Name));
71  	}
72  
73  	@Override
74  	public String findValue(final String Name) {
75  		if (vars.isEmpty() || !vars.containsKey(Name))
76  			return null;
77  		return vars.get(Name);
78  	}
79  
80  	@Override
81  	public String process(final String value) {
82  		final Matcher m = REGEX_VAR.matcher(value);
83  		if (!m.find())
84  			return value;
85  
86  		//We found a variable, so loop through all the matches
87  		//and do our find/replace.
88  		final StringBuffer sb = new StringBuffer(128);
89  		do {
90  			if (m.groupCount() < 1)
91  				continue;
92  
93  			String varName = m.group(1);
94  			String varValue = vars.get(varName);
95  			if (varValue == null)
96  				throw new MissingVariableException("Unable to locate variable: " + varName + ". Variable names are case-sensitive.");
97  			
98  			m.appendReplacement(sb, varValue);
99  		} while(m.find());
100 		m.appendTail(sb);
101 		
102 		return sb.toString();
103 	}
